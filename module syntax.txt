Every module should follow this shape:

js
// modules/gamification.js
export default {
  name: "gamification-axis",

  init(ctx) {
    console.log("[Gamification] init", ctx);

  onSignal(type, payload, ctx) {
    console.log("[Gamification] signal:", type, payload);
  },

  onStateChanged(state) {
